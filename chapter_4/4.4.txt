4-4 格式化输入与输出

4-4-1 格式化输出概论
FORMAT 命令可以用来设置输出格式

EX0420.F90
PROGRAM EX0420
INTEGER A
A=100
WRITE(*,100) A   !使用行代码100的格式来输出变量
100 FORMAT(I4)   ！最前面的100是行代码，I4表示只用4列来显示整数
END

EX0421.F90 
PROGRAM EX0421
INTEGER A
REAL    B
COMPLEX C
LOGICAL D
CHARACTER(LEN=20)   E 
A=10
B=12.0
C=(1.0,2.0)
D=.TRUE.
E="FORTRAN"
WRITE(*,100) A
100 FORMAT(I3)
WRITE(*,200) B
200 FORMAT(F4.2)
WRITE(*,300) C 
300 FORMAT(F4.1, F4.1)
WRITE(*,400) D
400 FORMAT(L3)
WRITE(*,500) E 
500 FORMAT(A10)
END
